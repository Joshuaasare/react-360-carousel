## Install

```bash
npm install --save react-360-carousel
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-360-carousel'
import 'react-360-carousel/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
